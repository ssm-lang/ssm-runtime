76 /** @brief Percolate act queue hole up, to find where to put a new entry.
77  *
78  * Starting at the hole, walk up toward the root of the tree, copying parent to
79  * child until we find where we can put the new activation record.
80  *
81  * @param hole index of the hole that needs to be filled.
82  * @param act activation record that needs to be placed in the heap.
83  */
84 SSM_STATIC_INLINE
85 void act_queue_percolate_up(q_idx_t hole, ssm_act_t *act) {
86  // GCOV_EXCL_START
87  SSM_ASSERT(hole >= SSM_QUEUE_HEAD && hole <= act_queue_len);
88  // GCOV_EXCL_STOP
89  ssm_priority_t priority = act->priority;
90  for (; hole > SSM_QUEUE_HEAD && priority < act_queue[hole >> 1]->priority;
91  hole >>= 1)
92  act_queue[hole] = act_queue[hole >> 1];
93  act_queue[hole] = act;
94  act->scheduled = true;
95 }

97 /** @brief Percolate act queue hole down, to find where to put a new entry.
98  *
99  * Starting at the hole, walk down towards the leaves of the tree, moving the
100  * earlier child to the parent and repeating the process on that child. This
101  * makes the parent earlier than both children. Stop when the activation record
102  * we're trying to place has a lower priority than either children.
103  *
104  * @param hole where to place the given activation record.
105  * @param act activation record to be placed in the queue.
106  */
107 SSM_STATIC_INLINE
108 void act_queue_percolate_down(q_idx_t hole, ssm_act_t *act) {
109  // GCOV_EXCL_START
110  SSM_ASSERT(hole >= SSM_QUEUE_HEAD && hole <= act_queue_len);
111  // GCOV_EXCL_STOP
112  ssm_priority_t priority = act->priority;
113  for (;;) {
114  // Find the earlier of the two children
115  q_idx_t child = hole << 1; // Left child
116  if (child > act_queue_len)
117  break; // The parent was a leaf
118  if (child + 1 <= act_queue_len &&
119  act_queue[child + 1]->priority < act_queue[child]->priority)
120  child++; // Right child is earlier than the left
121 
122  if (priority < act_queue[child]->priority)
123  break; // Earlier child is later than what we're inserting
124  act_queue[hole] = act_queue[child];
125  hole = child;
126  }
127  act_queue[hole] = act;
128 }

198  // Should be in the queue

215  return;

328 void ssm_later(ssm_sv_t *var, ssm_time_t later, ssm_value_t value) {
329  if (later < now)
330  // later must be in the future
331  SSM_THROW(SSM_INVALID_TIME);
332 
333  var->later_value = value;
334 
335  if (var->later_time == SSM_NEVER) {
336  // Variable does not have a pending event: add it to the queue
337  q_idx_t hole = ++event_queue_len;
338 
339  if (event_queue_len > SSM_EVENT_QUEUE_SIZE)
340  SSM_THROW(SSM_EXHAUSTED_EVENT_QUEUE);
341 
342  var->later_time = later;
343  event_queue_percolate_up(hole, var);
344 
345  } else {
346  // Variable has a pending event: reposition the event in the queue
347  // as appropriate
348  q_idx_t hole = find_queued_event(var);
349 
350  var->later_time = later;
351 
352  if (hole == SSM_QUEUE_HEAD || event_queue[hole >> 1]->later_time < later)
353  event_queue_percolate_down(hole, var);
354  else
355  event_queue_percolate_up(hole, var);
356  }
357 }

383 void ssm_unschedule(ssm_sv_t *var) {
384  if (var->later_time != SSM_NEVER) {
385  q_idx_t hole = find_queued_event(var);
386  var->later_time = SSM_NEVER;
387  ssm_sv_t *moved_var = event_queue[event_queue_len--];
388  if (hole < SSM_QUEUE_HEAD + event_queue_len) {
389  // Percolate only if removal led to a hole in the queue; no need to do
390  // this if we happened to remove the last element of the queue.
391  if (hole == SSM_QUEUE_HEAD ||
392  event_queue[hole >> 1]->later_time < moved_var->later_time)
393  event_queue_percolate_down(hole, moved_var);
394  else
395  event_queue_percolate_up(hole, moved_var);
396  }
397  }
398 }

433 void ssm_tick(void) {
434  if (act_queue_len != 0 && ssm_next_event_time() != SSM_NEVER &&
435  now > ssm_next_event_time())
436  // There are somehow events in the queue that have a timestamp earlier than
437  // the time at which we are trying to tick. This is usually a sign of queue
438  // mismanagement by platform code.
439  SSM_THROW(SSM_NOT_READY);
440 
441  if (act_queue_len == 0 && event_queue_len > 0)
442  ssm_set_now(ssm_next_event_time());
443 
444  // Update every variable in the event queue at the current time.
445  while (event_queue_len > 0 &&
446  event_queue[SSM_QUEUE_HEAD]->later_time == now) {
447  ssm_sv_t *sv = event_queue[SSM_QUEUE_HEAD];
448  ssm_update(sv);
449 
450  // Remove the top event from the queue by inserting the last element in the
451  // queue at the front and percolating it toward the leaves.
452  ssm_sv_t *to_insert = event_queue[event_queue_len--];
453 
454  if (event_queue_len) /* Only percolate if there are other events left */
455  event_queue_percolate_down(SSM_QUEUE_HEAD, to_insert);
456  }
457 
458  while (act_queue_len > 0) {
459  ssm_act_t *to_run = act_queue[SSM_QUEUE_HEAD];
460  to_run->scheduled = false;
461 
462  // Remove the top activation record from the queue by inserting the last
463  // element in the queue at the front and percolating it down.
464  ssm_act_t *to_insert = act_queue[act_queue_len--];
465 
466  if (act_queue_len)
467  act_queue_percolate_down(SSM_QUEUE_HEAD, to_insert);
468 
469  to_run->step(to_run);
470  }
471 }
