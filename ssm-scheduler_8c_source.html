76 /** @brief Percolate act queue hole up, to find where to put a new entry.
77  *
78  * Starting at the hole, walk up toward the root of the tree, copying parent to
79  * child until we find where we can put the new activation record.
80  *
81  * @param hole index of the hole that needs to be filled.
82  * @param act activation record that needs to be placed in the heap.
83  */
84 SSM_STATIC_INLINE
85 void act_queue_percolate_up(q_idx_t hole, ssm_act_t *act) {
86  // GCOV_EXCL_START
87  SSM_ASSERT(hole >= SSM_QUEUE_HEAD && hole <= act_queue_len);
88  // GCOV_EXCL_STOP
89  ssm_priority_t priority = act->priority;
90  for (; hole > SSM_QUEUE_HEAD && priority < act_queue[hole >> 1]->priority;
91  hole >>= 1)
92  act_queue[hole] = act_queue[hole >> 1];
93  act_queue[hole] = act;
94  act->scheduled = true;
95 }

97 /** @brief Percolate act queue hole down, to find where to put a new entry.
98  *
99  * Starting at the hole, walk down towards the leaves of the tree, moving the
100  * earlier child to the parent and repeating the process on that child. This
101  * makes the parent earlier than both children. Stop when the activation record
102  * we're trying to place has a lower priority than either children.
103  *
104  * @param hole where to place the given activation record.
105  * @param act activation record to be placed in the queue.
106  */
107 SSM_STATIC_INLINE
108 void act_queue_percolate_down(q_idx_t hole, ssm_act_t *act) {
109  // GCOV_EXCL_START
110  SSM_ASSERT(hole >= SSM_QUEUE_HEAD && hole <= act_queue_len);
111  // GCOV_EXCL_STOP
112  ssm_priority_t priority = act->priority;
113  for (;;) {
114  // Find the earlier of the two children
115  q_idx_t child = hole << 1; // Left child
116  if (child > act_queue_len)
117  break; // The parent was a leaf
118  if (child + 1 <= act_queue_len &&
119  act_queue[child + 1]->priority < act_queue[child]->priority)
120  child++; // Right child is earlier than the left
121 
122  if (priority < act_queue[child]->priority)
123  break; // Earlier child is later than what we're inserting
124  act_queue[hole] = act_queue[child];
125  hole = child;
126  }
127  act_queue[hole] = act;
128 }

187 /** @brief Determine the index of an already scheduled event.
188  *
189  * This is an inefficient linear search, so it's not meant to be used often.

192  * queue.
193  *
194  * @param var the variable being searched for.

196 SSM_STATIC_INLINE
197 q_idx_t find_queued_event(ssm_sv_t *var) {
198  // Should be in the queue
199  // GCOV_EXCL_START (control flow here is too predictable to warrant coverage)
200  SSM_ASSERT(var->later_time != SSM_NEVER);
201  for (q_idx_t i = SSM_QUEUE_HEAD; i <= event_queue_len; i++)
202  if (event_queue[i] == var)
203  return i;

205  return 0;

309 void ssm_later(ssm_sv_t *var, ssm_time_t later, ssm_value_t value) {
310  if (later < now)
311  // later must be in the future
312  SSM_THROW(SSM_INVALID_TIME);
313 
314  var->later_value = value;
315 
316  if (var->later_time == SSM_NEVER) {
317  // Variable does not have a pending event: add it to the queue
318  q_idx_t hole = ++event_queue_len;
319 
320  if (event_queue_len > SSM_EVENT_QUEUE_SIZE)
321  SSM_THROW(SSM_EXHAUSTED_EVENT_QUEUE);
322 
323  var->later_time = later;
324  event_queue_percolate_up(hole, var);
325 
326  } else {
327  // Variable has a pending event: reposition the event in the queue
328  // as appropriate
329  q_idx_t hole = find_queued_event(var);
330 
331  var->later_time = later;
332 
333  if (hole == SSM_QUEUE_HEAD || event_queue[hole >> 1]->later_time < later)
334  event_queue_percolate_down(hole, var);
335  else
336  event_queue_percolate_up(hole, var);
337  }
338 }

364 void ssm_unschedule(ssm_sv_t *var) {
365  if (var->later_time != SSM_NEVER) {
366  q_idx_t hole = find_queued_event(var);
367  var->later_time = SSM_NEVER;
368  ssm_sv_t *moved_var = event_queue[event_queue_len--];
369  if (hole < SSM_QUEUE_HEAD + event_queue_len) {
370  // Percolate only if removal led to a hole in the queue; no need to do
371  // this if we happened to remove the last element of the queue.
372  if (hole == SSM_QUEUE_HEAD ||
373  event_queue[hole >> 1]->later_time < moved_var->later_time)
374  event_queue_percolate_down(hole, moved_var);
375  else
376  event_queue_percolate_up(hole, moved_var);
377  }
378  }
379 }

414 void ssm_tick(void) {
415  if (act_queue_len != 0 && ssm_next_event_time() != SSM_NEVER &&
416  now > ssm_next_event_time())
417  // There are somehow events in the queue that have a timestamp earlier than
418  // the time at which we are trying to tick. This is usually a sign of queue
419  // mismanagement by platform code.
420  SSM_THROW(SSM_NOT_READY);
421 
422  if (act_queue_len == 0 && event_queue_len > 0)
423  ssm_set_now(ssm_next_event_time());
424 
425  // Update every variable in the event queue at the current time.
426  while (event_queue_len > 0 &&
427  event_queue[SSM_QUEUE_HEAD]->later_time == now) {
428  ssm_sv_t *sv = event_queue[SSM_QUEUE_HEAD];
429  ssm_update(sv);
430 
431  // Remove the top event from the queue by inserting the last element in the
432  // queue at the front and percolating it toward the leaves.
433  ssm_sv_t *to_insert = event_queue[event_queue_len--];
434 
435  if (event_queue_len) /* Only percolate if there are other events left */
436  event_queue_percolate_down(SSM_QUEUE_HEAD, to_insert);
437  }
438 
439  while (act_queue_len > 0) {
440  ssm_act_t *to_run = act_queue[SSM_QUEUE_HEAD];
441  to_run->scheduled = false;
442 
443  // Remove the top activation record from the queue by inserting the last
444  // element in the queue at the front and percolating it down.
445  ssm_act_t *to_insert = act_queue[act_queue_len--];
446 
447  if (act_queue_len)
448  act_queue_percolate_down(SSM_QUEUE_HEAD, to_insert);
449 
450  to_run->step(to_run);
451  }
452 }
