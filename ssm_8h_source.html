307 /** @brief A scheduled variable that supported scheduled updates with triggers.
308  *
309  * Scheduled variables are heap-allocated built-in types that represent
310  * variables with reference-like semantics in SSM. This struct only represents
311  * the data specific to scheduled variables, and is not meant to be declared on
312  * its own. Instead, it should be allocated as part of an #ssm_mm using
313  * ssm_new().
314  *
315  * Routines may directly assign to them in the current instant (ssm_assign()),
316  * or schedule a delayed assignment to them (ssm_later()). The @a last_updated
317  * time is recorded in each case, sensitive routines, i.e., @a triggers, are
318  * woken up.
319  *
320  * At most one delayed assignment may be scheduled at a time, but a single
321  * update may wake any number of sensitive routines.
322  *
323  * @invariant @a later_time != #SSM_NEVER iff this variable in the event queue.
324  * @invariant for all `ssm_sv_t v`, `ssm_mm_is(SSM_SV_T, &v.mm)`.
325  */
326 typedef struct ssm_sv {
327  struct ssm_mm mm;
328  ssm_time_t later_time; /**< When the variable should be next updated. */
329  ssm_time_t last_updated; /**< When the variable was last updated. */
330  ssm_trigger_t *triggers; /**< List of sensitive continuations. */
331  ssm_value_t value; /**< Current value. */
332  ssm_value_t later_value; /**< Buffered value for delayed assignment. */
333 } ssm_sv_t;
334 
335 /** @brief Initial value of an #ssm_sv.
336  *
337  * For example:
338  *
339  * ssm_value_t sv;
340  * sv.heap_ptr = ssm_new(SSM_BUILTIN, SSM_SV_T);
341  * ssm_sv_init(sv, ssm_marshal(0));
342  *
343  * @note Initialization does not count as an update event, so @a last_updated
344  * is initialized to #SSM_NEVER.
345  * @note @a later_value is left uninitialized.
346  *
347  * @param sv pointer to the scheduled variable.
348  * @param val the initial value of the schedueld variable.
349  */
350 #define ssm_sv_init(sv, val) \
351  do { \
352  ssm_to_sv(sv)->later_time = SSM_NEVER; \
353  ssm_to_sv(sv)->last_updated = SSM_NEVER; \
354  ssm_to_sv(sv)->triggers = NULL; \
355  ssm_to_sv(sv)->value = val; \
