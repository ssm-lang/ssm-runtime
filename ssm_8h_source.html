224 /** @brief The memory management metadata "header" for heap-allocated objects.
225  *
226  * This header should always be embedded in heap-allocated objects as the first
227  * field (at memory offset 0); values of type #ssm_value_t will point to this
228  * header and use its @a kind and other fields to figure out the size and
229  * memory layout of the rest of the object.
230  *
231  * The interpretation and usage of the latter 16 bits of this header, i.e., the
232  * @a info field, depends on the value of @a kind. For objects encoding
233  * variants (e.g., #ssm_adt1), the @a count field counts the number of fields
234  * in the object payload, while the @a tag records which variant is inhabited
235  * by the object. Meanwhile, vector-style objects (e.g., #ssm_closure1) also
236  * use the @a count field to record the number of values present, but use the
237  * @a cap field to determine the full capacity of the payload. Finally, objects
238  * like arrays and blobs do not need to record more than their size, and
239  * benefit from making use of all 16 bits to support up to 65536 sizes.
240  */
241 struct ssm_mm {
242  uint8_t ref_count; /**< The number of references to this object. */
243  uint8_t kind; /**< The #ssm_kind of object this is. */
244  union {
245  struct {
246  uint8_t count; /**< Number of #ssm_value_t values in payload. */
247  uint8_t tag; /**< Which variant is inhabited by this object. */
248  } variant;
249  struct {
250  uint8_t count; /**< Number of #ssm_value_t values in payload. */
251  uint8_t cap; /**< Which variant is inhabited by this object. */
252  } vector;
253  uint16_t size; /**< 16-bit size */
254  } info; /**< Three "flavors" of information embedded in the header. */
255 };

540 /** @brief A scheduled variable that supports scheduled updates with triggers.
541  *
542  * Scheduled variables are heap-allocated built-in types that represent
543  * variables with reference-like semantics in SSM. These should be always
544  * allocated on the heap using ssm_new_sv().
545  *
546  * Routines may directly assign to them in the current instant (ssm_assign()),
547  * or schedule a delayed assignment to them (ssm_later()). The @a last_updated
548  * time is recorded in each case, sensitive routines, i.e., @a triggers, are
549  * woken up.

552  * update may wake any number of sensitive routines.
553  *

555  * @invariant for all `ssm_sv_t v`, `v.mm.kind == SSM_SV_K`.
556  */

558  struct ssm_mm mm;
559  ssm_time_t later_time; /**< When the variable should be next updated. */
560  ssm_time_t last_updated; /**< When the variable was last updated. */
561  ssm_trigger_t *triggers; /**< List of sensitive continuations. */
562  ssm_value_t value; /**< Current value. */
563  ssm_value_t later_value; /**< Buffered value for delayed assignment. */
564 } ssm_sv_t;
