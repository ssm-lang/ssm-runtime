500 /** @brief A scheduled variable that supported scheduled updates with triggers.
501  *
502  * Scheduled variables are heap-allocated structures that represent variables
503  * with reference-like semantics in SSM.
504  *
505  * Routines may directly assign to them in the current instant (ssm_assign()),
506  * or schedule a delayed assignment to them (ssm_later()). The @a last_updated
507  * time is recorded in each case, sensitive routines, i.e., @a triggers, are
508  * woken up.
509  *
510  * At most one delayed assignment may be scheduled at a time, but a single
511  * update may wake any number of sensitive routines.
512  *
513  * @invariant @a later_time != #SSM_NEVER iff this variable in the event queue.
514  * @invariant For all `struct ssm_time t`, `ssm_mm_is_builtin(SSM_SV_T, t)`.
515  */
516 typedef struct ssm_sv {
517  struct ssm_mm mm; /**< Memory management metadata. */
518  ssm_time_t later_time; /**< When the variable should be next updated. */
519  ssm_time_t last_updated; /**< When the variable was last updated. */
520  ssm_trigger_t *triggers; /**< List of sensitive continuations. */
521  ssm_value_t value; /**< Current value. */
522  ssm_value_t later_value; /**< Buffered value for delayed assignment. */
523 } ssm_sv_t;
