422 /** @brief A scheduled variable that supports scheduled updates with triggers.
423  *
424  * Scheduled variables are heap-allocated built-in types that represent
425  * variables with reference-like semantics in SSM. These should be always
426  * allocated on the heap using ssm_new_sv().
427  *
428  * Routines may directly assign to them in the current instant (ssm_assign()),
429  * or schedule a delayed assignment to them (ssm_later()). The @a last_updated
430  * time is recorded in each case, sensitive routines, i.e., @a triggers, are
431  * woken up.
432  *
433  * At most one delayed assignment may be scheduled at a time, but a single
434  * update may wake any number of sensitive routines.
435  *
436  * @invariant @a later_time != #SSM_NEVER iff this variable in the event queue.
437  * @invariant for all `ssm_sv_t v`, `v.mm.kind == SSM_SV_K`.
438  */
439 typedef struct ssm_sv {
440  struct ssm_mm mm;
441  ssm_time_t later_time; /**< When the variable should be next updated. */
442  ssm_time_t last_updated; /**< When the variable was last updated. */
443  ssm_trigger_t *triggers; /**< List of sensitive continuations. */
444  ssm_value_t value; /**< Current value. */
445  ssm_value_t later_value; /**< Buffered value for delayed assignment. */
446 } ssm_sv_t;

482 /** @brief Instantaneous assignment to a scheduled variable.
483  *
484  * Updates the value of @a var in the current instant, and wakes up all
485  * sensitive processes at a lower priority than the calling routine.
486  *
487  * ssm_assign() will ssm_drop() the previous value of @a var and ssm_dup() the
488  * new @a val. If the caller already knows whether @a var and @a val reside on
489  * the heap, they may call ssm_assign_unsafe() to forego the ssm_drop() and
490  * ssm_dup() calls, but will be responsible for reference counting themselves
491  * (i.e., calling ssm_dup_unsafe() and ssm_drop_unsafe() for heap objects).
492  *
493  * @note Does not overwrite a scheduled assignment.
494  * @note The behavior of this macro when @a var does not point to a scheduled
495  * variable is undefined.
496  *
497  * @param var pointer to the scheduled variable.
498  * @param prio priority of the calling routine.
499  * @param val the value to be assigned to @a var.
500  */

502  do { \

506  } while (0)

796 /** @brief Apply an argument to a closure.

802  * must yield to the scheduler to allow the child process to run. The child
803  * process will write its return value to @a ret.

806  * @a arg pushed to its @a argv will be written to @a ret.
