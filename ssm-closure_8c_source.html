23 ssm_value_t ssm_closure_clone(ssm_value_t closure) {
24  ssm_value_t new_closure =

26  ssm_closure_arg_count(new_closure) = ssm_closure_arg_count(closure);

29  ssm_dup(ssm_closure_arg(closure, i));
30  ssm_closure_arg(new_closure, i) = ssm_closure_arg(closure, i);
31  }

36 void ssm_closure_apply(ssm_value_t closure, ssm_value_t arg, ssm_act_t *parent,
37  ssm_priority_t prio, ssm_depth_t depth,
38  ssm_value_t *ret) {
39  if (ssm_closure_arg_cap(closure) == ssm_closure_arg_count(closure) + 1) {
40  // Call ssm_dup() on all previously applied arguments of closure, since they
41  // will be shared by the child process.
42  ssm_dups(ssm_closure_arg_count(closure), ssm_closure_argv(closure));
43  ssm_closure_push(closure, arg);
44  ssm_closure_activate(closure, parent, prio, depth, ret);
45  ssm_closure_pop(closure);
46  } else {
47  // No need to call ssm_dups() since ssm_closure_clone() already dups
48  // all the arguments.
49  *(ret) = ssm_closure_clone(closure);
50  ssm_closure_push(*(ret), arg);
51  }
52 }

54 void ssm_closure_apply_final(ssm_value_t closure, ssm_value_t arg,
55  ssm_act_t *parent, ssm_priority_t prio,
56  ssm_depth_t depth, ssm_value_t *ret) {
57  SSM_ASSERT(!ssm_is_shared(closure));
58 
59  if (ssm_closure_arg_cap(closure) == ssm_closure_arg_count(closure) + 1) {
60  ssm_closure_push(closure, arg);
61  ssm_closure_activate(closure, parent, prio, depth, ret);
62 
63  // We can directly call ssm_mem_free() because we already know (1) closure
64  // is a heap object and (2) this is the last time it will be used.
65  //
66  // We don't need to call ssm_drop() on any of its arguments since those are
67  // moved into the newly activated process (also why we didn't need to call
68  // ssm_dup() before activating the process).
69  ssm_closure_free(closure);
70  } else {
71  // Just modify the closure in-place.
72  ssm_closure_push(closure, arg);
73  *(ret) = (closure);
