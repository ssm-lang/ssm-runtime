13 /** @brief (The beginning of) a block of memory.
14  *
15  * The size of each block varies depending on which memory pool it resides in.
16  * When a block is being used (allocated), @a block_buf effectively gives
17  * a pointer to the beginning that block that can be indexed into.
18  *
19  * When a block is not being used, it is maintained in a <em>free list</em>.
20  * If it is not at the head of the free list, then it is pointed to by the @a
21  * free_list_next field of some other block_t, while its own @a free_list_next
22  * points to the next block in the free list. To avoid initializing the @a
23  * free_list_next fields of "fresh" blocks, when @a free_list_next is
24  * #UNINITIALIZED_FREE_BLOCK, the next free block contiguously follows the
25  * current block in memory. The last free block points to #END_OF_FREELIST.
26  */
27 typedef union block {
28  union block *free_list_next; /**< Pointer to the next block. */
29  uint8_t block_buf[1]; /**< Variable-size buffer of the block. */
30 } block_t;

62 /** @brief Find the memory pool for some arbitrary size.
63  *
64  * Tries to find the smallest memory pool that will fit a block of @a size.
65  * Returns #SSM_MEM_POOL_COUNT if no such pool exists, i.e., if @a size is
66  * greater than #SSM_MEM_POOL_MAX.
67  *
68  * Since #SSM_MEM_POOL_COUNT is a (small) compile-time constant, this linear
69  * search is effectively contant time.
70  *
71  * @param size the block size whose memory pool we are looking for.
72  * @returns index to the memory pool, or #SSM_MEM_POOL_COUNT otherwise.
73  */
74 static inline size_t find_pool_size(size_t size) {
75  for (size_t pool_idx = 0; pool_idx < SSM_MEM_POOL_COUNT; pool_idx++)
76  if (size < SSM_MEM_POOL_SIZE(pool_idx))
77  return pool_idx;
78  return SSM_MEM_POOL_COUNT;
79 }

81 /** @brief Allocate a new block for a memory pool.
82  *
83  * Calls alloc_page() to allocate a new zero-initialized page for the memory
84  * pool, and adds it to the corresponding memory pool with stack discipline.
85  * The last block of the free list is pointed to the old head of the free list.
86  *
87  * Constant time (not counting what alloc_page() does).
88  *
89  * @param p the index of the memory pool.
90  */
91 static inline void alloc_pool(size_t p) {
92  block_t *new_page = alloc_page();
93  SSM_ASSERT(END_OF_FREELIST < new_page);
94 
95  struct mem_pool *pool = &mem_pools[p];
96  size_t last_block = BLOCKS_PER_PAGE - SSM_MEM_POOL_SIZE(p) / sizeof(block_t);
97  new_page[last_block].free_list_next = pool->free_list_head;
98  pool->free_list_head = new_page;
99 }

120 void *ssm_mem_alloc(size_t size) {
121 #ifdef SSM_DEBUG_NO_ALLOC
122  return alloc_mem(size);
123 #else
124  size_t p = find_pool_size(size);
125  if (p >= SSM_MEM_POOL_COUNT)
126  return alloc_mem(size);
127 
128  struct mem_pool *pool = &mem_pools[p];
129 
130  if (pool->free_list_head == END_OF_FREELIST)
131  alloc_pool(p);
132 
133  void *buf = pool->free_list_head->block_buf;
134 
135  if (pool->free_list_head->free_list_next == UNINITIALIZED_FREE_BLOCK)
136  pool->free_list_head += SSM_MEM_POOL_SIZE(p) / sizeof(block_t);
137  else
138  pool->free_list_head = pool->free_list_head->free_list_next;
139 
140  return buf;
141 #endif
142 }

144 void ssm_mem_free(void *m, size_t size) {
145 #ifdef SSM_DEBUG_NO_ALLOC
146  free_mem(m, size);
147 #else
148  size_t pool = find_pool_size(size);
149  if (pool >= SSM_MEM_POOL_COUNT) {
150  free_mem(m, size);
151  return;
152  }
153 
154  block_t *new_head = m;
155  new_head->free_list_next = mem_pools[pool].free_list_head;
156  mem_pools[pool].free_list_head = new_head;
157 #endif
158 }

188 void ssm_drop_final(ssm_value_t v) {
189  size_t size = 0;
190  switch (v.heap_ptr->kind) {
191  case SSM_ADT_K:
192  size = ssm_adt_size(v.heap_ptr->val_count);
193  for (size_t i = 0; i < v.heap_ptr->val_count; i++)
194  ssm_drop(ssm_adt_field(v, i));
195  break;
196  case SSM_TIME_K:
197  size = sizeof(struct ssm_time);
198  break;
199  case SSM_SV_K:
200  size = sizeof(struct ssm_sv);
201  ssm_unschedule(ssm_to_sv(v));
202  ssm_drop(ssm_to_sv(v)->value);
203  if (ssm_to_sv(v)->later_time != SSM_NEVER)
204  ssm_drop(ssm_to_sv(v)->later_value);
205  break;
206  }
207  ssm_mem_free(v.heap_ptr, size);
208 }
