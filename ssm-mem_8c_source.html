13 /** @brief (The beginning of) a block of memory.
14  *
15  * The size of each block varies depending on which memory pool it resides in.
16  * When a block is being used (allocated), @a block_buf effectively gives
17  * a pointer to the beginning that block that can be indexed into.
18  *
19  * When a block is not being used, it is maintained in a <em>free list</em>.
20  * If it is not at the head of the free list, then it is pointed to by the @a
21  * free_list_next field of some other block_t, while its own @a free_list_next
22  * points to the next block in the free list. To avoid initializing the @a
23  * free_list_next fields of "fresh" blocks, when @a free_list_next is
24  * #UNINITIALIZED_FREE_BLOCK, the next free block contiguously follows the
25  * current block in memory. The last free block points to #END_OF_FREELIST.
26  */
27 typedef union block {
28  union block *free_list_next; /**< Pointer to the next block. */
29  uint8_t block_buf[1]; /**< Variable-size buffer of the block. */
30 } block_t;

62 /** @brief Find the memory pool for some arbitrary size.
63  *
64  * Tries to find the smallest memory pool that will fit a block of @a size.
65  * Returns #SSM_MEM_POOL_COUNT if no such pool exists, i.e., if @a size is
66  * greater than #SSM_MEM_POOL_MAX.
67  *
68  * Since #SSM_MEM_POOL_COUNT is a (small) compile-time constant, this linear
69  * search is effectively contant time.
70  *
71  * @param size the block size whose memory pool we are looking for.
72  * @returns index to the memory pool, or #SSM_MEM_POOL_COUNT otherwise.
73  */
74 static inline size_t find_pool_size(size_t size) {
75  for (size_t pool_idx = 0; pool_idx < SSM_MEM_POOL_COUNT; pool_idx++)
76  if (size < SSM_MEM_POOL_SIZE(pool_idx))
77  return pool_idx;
78  return SSM_MEM_POOL_COUNT;
79 }

81 static inline block_t *find_next_block(block_t *block, size_t pool_size) {
82  if (block->free_list_next == UNINITIALIZED_FREE_BLOCK)
83  return block + pool_size / sizeof(block_t);
84  else
85  return block->free_list_next;
86 }

88 /** @brief Allocate a new block for a memory pool.
89  *
90  * Calls alloc_page() to allocate a new zero-initialized page for the memory
91  * pool, and adds it to the corresponding memory pool with stack discipline.
92  * The last block of the free list is pointed to the old head of the free list.
93  *
94  * Constant time (not counting what alloc_page() does).
95  *
96  * @param p the index of the memory pool.
97  */
98 static inline void alloc_pool(size_t p) {
99  block_t *new_page = alloc_page();
100  SSM_ASSERT(END_OF_FREELIST < new_page);
101  struct mem_pool *pool = &mem_pools[p];
102  size_t last_block = BLOCKS_PER_PAGE - SSM_MEM_POOL_SIZE(p) / sizeof(block_t);
103  new_page[last_block].free_list_next = pool->free_list_head;
104  pool->free_list_head = new_page;
105 
106 #ifdef USE_VALGRIND
107  // Mark as NOACCESS; nothing should touch this memory until it is allocated.
108  VALGRIND_MAKE_MEM_NOACCESS(new_page, SSM_MEM_PAGE_SIZE);
109 
110  // Label this page with a human-readable name, for memory error reporting.
111  VALGRIND_CREATE_BLOCK(new_page, SSM_MEM_PAGE_SIZE, "SSM memory pool page");
112 #endif
113 }

160 void *ssm_mem_alloc(size_t size) {
161  size_t p = find_pool_size(size);
162  if (p >= SSM_MEM_POOL_COUNT)
163  return alloc_mem(size);
164 
165  struct mem_pool *pool = &mem_pools[p];
166 
167  if (pool->free_list_head == END_OF_FREELIST)
168  alloc_pool(p);
169 
170  void *m = pool->free_list_head->block_buf;
171 
172 #ifdef USE_VALGRIND
173  // Tell Valgrind that we have allocated m as a chunk of pool.
174  //
175  // The memory range [m..m+size] is now considered DEFINED.
176  VALGRIND_MALLOCLIKE_BLOCK(m, size, 0, 1);
177 #endif
178 
179  pool->free_list_head =
180  find_next_block(pool->free_list_head, SSM_MEM_POOL_SIZE(p));
181 
182 #ifdef USE_VALGRIND
183  // Make the memory range [m..m+size] undefined, because the caller should
184  // not rely on allocated chunks being defined.
185  VALGRIND_MAKE_MEM_UNDEFINED(m, size);
186 #endif
187 
188  return m;
189 }

191 void ssm_mem_free(void *m, size_t size) {
192  size_t p = find_pool_size(size);
193  if (p >= SSM_MEM_POOL_COUNT) {
194  free_mem(m, size);
195  return;
196  }
197 
198  struct mem_pool *pool = &mem_pools[p];
199 
200  block_t *new_head = m;
201  new_head->free_list_next = pool->free_list_head;
202  pool->free_list_head = new_head;
203 
204 #ifdef USE_VALGRIND
205  // Tell Valgrind that we have freed m.
206  VALGRIND_FREELIKE_BLOCK(m, 0);
207 #endif
208 }

238 void ssm_drop_final(ssm_value_t v) {
239  size_t size = 0;
240  switch (v.heap_ptr->kind) {
241  case SSM_ADT_K:
242  size = ssm_adt_size(v.heap_ptr->val_count);
243  for (size_t i = 0; i < v.heap_ptr->val_count; i++)
244  ssm_drop(ssm_adt_field(v, i));
245  break;
246  case SSM_TIME_K:
247  size = sizeof(struct ssm_time);
248  break;
249  case SSM_SV_K:
250  size = sizeof(struct ssm_sv);
251  ssm_unschedule(ssm_to_sv(v));
252  ssm_drop(ssm_to_sv(v)->value);
253  if (ssm_to_sv(v)->later_time != SSM_NEVER)
254  ssm_drop(ssm_to_sv(v)->later_value);
255  break;
256  case SSM_CLOSURE_K:
257  size = ssm_closure_size(ssm_closure_arg_cap(v));
258  for (size_t i = 0; i < ssm_closure_arg_count(v); i++)
259  ssm_drop(ssm_closure_arg(v, i));
260  break;
261  }
262  ssm_mem_free(v.heap_ptr, size);
263 }
